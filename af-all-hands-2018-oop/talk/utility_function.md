```ruby
require 'forwardable'

class Attendees
  extend Forwardable

  def initialize(collection)
    @collection = collection
  end

  def excited?
    excited, indifferent = partition(&:excited?)
    excited.count > indifferent.count
  end

  delegate partition: :@collection
end
```
